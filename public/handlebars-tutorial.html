2Basic Syntax

{{! Basic variable output }}
<h1>Hello, {{name}}!</h1>

{{! Conditional rendering }}
{{#if user}}
    Welcome back, {{user.name}}!
{{else}}
    Please log in.
{{/if}}

{{! Array iteration }}
{{#each items}}
    <li>{{this}}</li>
{{/each}}

Key Concepts:

  • {{variable}}: Output escaped content
  • {{{triple-stache}}}: Output unescaped HTML
  • {{#if}}: Conditional blocks
  • {{#each}}: Array iteration

3Express.js Integration

// Setup Handlebars in Express
const express = require('express');
const hbs = require('hbs');
const app = express();

// Configure view engine
app.set('view engine', 'hbs');
app.set('views', './views');

// Register partials directory
hbs.registerPartials('./views/partials');

4Custom Helpers

// Register custom helpers
hbs.registerHelper('formatDate', function(date) {
    return new Date(date).toLocaleDateString();
});

hbs.registerHelper('uppercase', function(str) {
    return str.toUpperCase();
});

// Use in templates
<p>Posted on: {{formatDate post.date}}</p>
<h1>{{uppercase title}}</h1>
